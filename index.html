Valorant cards

from pyodide.http import pyfetch import asyncio import js response = await pyfetch(url="https://valorant-api.com/v1/agents", method="GET") output = await response.json() agentes = [] for valor in output['data']: agentes.append(valor) nome = agentes[0]['displayName'] image = agentes[0]['fullPortrait'] background = agentes[0]['background'] card = document.getElementById('card') for agente in agentes: if(agente['fullPortrait'] == None): continue else: personagem = agente['displayName'] div_container = document.createElement("div") div_container.setAttribute("class", "container") div_container.setAttribute("id", f"container_id_{personagem}") card.append(div_container) box = document.getElementById(f'container_id_{personagem}') div_box = document.createElement("div") div_box.setAttribute("class", "box") div_box.setAttribute("id", f"{agente['displayName']}") box.append(div_box) imagem_agente = agente['fullPortrait'] elemento_imagem = document.createElement("img") elemento_imagem.setAttribute("src", f"{imagem_agente}") id_agente = document.getElementById(f'{personagem}') id_agente.append(elemento_imagem) background_agente = agente['background'] id_agente.style.backgroundImage = f"url('{background_agente}'), linear-gradient(#273753, red)"
